<script>

/* Floating hearts */

setInterval(()=>{

const heart=document.createElement("div");

heart.className="heart";

heart.innerHTML="💜";

heart.style.left=Math.random()*100+"vw";

heart.style.animationDuration=(Math.random()*3+3)+"s";

document.body.appendChild(heart);

setTimeout(()=>heart.remove(),6000);

},600);

/* Navigation */

function showGame(){landing.classList.remove("active");game.classList.add("active");}

function yesClicked(){game.classList.remove("active");passwordSection.classList.add("active");}

function enterDashboard(){countdownSection.classList.remove("active");dashboard.classList.add("active");startTypewriter();}

function showGiftPassword(){dashboard.classList.remove("active");giftPasswordSection.classList.add("active");}

function moveNo(){noBtn.style.position="absolute";noBtn.style.top=Math.random()*70+"%";noBtn.style.left=Math.random()*70+"%";}

/* Passwords */

const correctPassword="030403";

const giftPassword="B86172844";

function checkPassword(){

if(passwordInput.value===correctPassword){

passwordSection.classList.remove("active");

countdownSection.classList.add("active");

}else{wrongPass.innerText="Wrong password 😭";}

}

function unlockGift(){

if(giftPasswordInput.value===giftPassword){

giftPasswordSection.classList.remove("active");

giftSection.classList.add("active");

animateLayers();

}else{giftWrong.innerText="Incorrect waybill number 💔";}

}

/* Countdown */

const meetDate=new Date("May 15, 2026").getTime();

setInterval(()=>{

const now=new Date().getTime();

const d=Math.floor((meetDate-now)/(1000*60*60*24));

countdown.innerHTML=d+" days left 💜";

},1000);

/* Typewriter message */

const message=`Ayushmaaa,

'Kisi cheez ko agar poori shiddat se chaho toh poori qaayanat ussey tumse milane ki saazish mein jut jaati hain' - Truest of words....

I still don't believe that I am writing this Valentine's message for you :)

You have forever been the girl of my dreams!

The way you are, the way you think, the way you smile; everything is just perfect!

I am lucky enough to have been given this second chance by GOD (Fav hoon na unka) to share this life of mine with you.

Thank you so much for always showing faith in me please keep believing in me and I will keep proving how genuine your trust is.

Stay blessed and happy(That toh I will keep you only)

Hope this brings a smile to your GAANDU face :)

Happy Valentines' Ayushmaaaaaaa 💜`;

let i=0;

function startTypewriter(){

if(i<message.length){

typewriter.innerHTML+=message.charAt(i);

i++;

setTimeout(startTypewriter,30);

}

}

/* Animate gift layers */

function animateLayers(){

const layers=document.querySelectorAll(".layer");

layers.forEach((layer,index)=>{

setTimeout(()=>layer.classList.add("show"),index*800);

});

}
